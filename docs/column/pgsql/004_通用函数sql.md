```sql


----------------------------------检查函数的里包含的字符串-------------------------------------
SELECT n.nspname || '.' || p.proname AS function_name,
       p.prosrc                      AS function_source_code
FROM pg_proc p
         JOIN
     pg_namespace n ON n.oid = p.pronamespace
WHERE prosrc LIKE '%fish_latin%' and  prosrc not LIKE '%taxon_id%'
  and n.nspname = 'pkg_admin_v2';

SELECT n.nspname || '.' || p.proname AS function_name,
       p.prosrc                      AS function_source_code
FROM pg_proc p
         JOIN
     pg_namespace n ON n.oid = p.pronamespace
WHERE prosrc LIKE '%species_gather%'
  and n.nspname = 'pkg_api_v2';




-------------------查找依赖该函数的对象-----------------------------------------------------------
SELECT n.nspname || '.' || p.proname AS dependent_function
FROM pg_depend d
         JOIN
     pg_proc p ON p.oid = d.objid
         JOIN
     pg_namespace n ON n.oid = p.pronamespace
WHERE p.proname = 'pkg_taxon.observations_preferred_media' -- 替换为你的函数名
  AND d.deptype = 'n';


---------------------------------------查看正在运行的函数------------------------------------------
SELECT pid, usename, application_name, state, query
FROM pg_stat_activity
WHERE state = 'active'


--------------------------------查拥有某个字段的表-------------------------
--拥有fish_latin字段  但是不拥有taxon_id的表
SELECT
    c.table_schema,
    c.table_name,
    t.table_type
FROM information_schema.columns c
         JOIN information_schema.tables t
              ON c.table_schema = t.table_schema
                  AND c.table_name = t.table_name
GROUP BY
    c.table_schema,
    c.table_name,
    t.table_type
HAVING
    SUM(CASE WHEN c.column_name = 'fish_latin' THEN 1 ELSE 0 END) > 0
   AND SUM(CASE WHEN c.column_name = 'taxon_id' THEN 1 ELSE 0 END) = 0
ORDER BY
    c.table_schema,
    c.table_name;

SELECT
    n.nspname || '.' || p.proname AS function_name,
    p.prosrc AS function_source_code
FROM
    pg_proc p
        JOIN
    pg_namespace n ON n.oid = p.pronamespace
WHERE
    prosrc LIKE '%menu_fk%';

--查看正在运行的函数
SELECT pid, usename, application_name, state, query
FROM pg_stat_activity
WHERE state = 'active';

--取消查询
SELECT pg_cancel_backend(pid)
FROM pg_stat_activity
WHERE pid = <pid>;

--终止查询
SELECT pg_terminate_backend(pid)
FROM pg_stat_activity
WHERE pid = <pid>;






-- 或者更详细信息，包括表类型（普通表还是 foreign table）
SELECT table_schema,
       table_name,
       table_type
FROM information_schema.tables
WHERE table_name = 'solot_express';

SELECT ft.foreign_table_name,
       fs.foreign_server_name
FROM information_schema.foreign_tables ft
         JOIN
     information_schema.foreign_servers fs
     ON
         ft.foreign_server_name = fs.foreign_server_name
WHERE ft.foreign_table_name = 'solot_express';
```
